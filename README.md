const puzzle = [
[5,3,0,0,7,0,0,0,0],
[6,0,0,1,9,5,0,0,0],
[0,9,8,0,0,0,0,6,0],

[8,0,0,0,6,0,0,0,3],
[4,0,0,8,0,3,0,0,1],
[7,0,0,0,2,0,0,0,6],

[0,6,0,0,0,0,2,8,0],
[0,0,0,4,1,9,0,0,5],
[0,0,0,0,8,0,0,7,9]
];

const solution = [
[5,3,4,6,7,8,9,1,2],
[6,7,2,1,9,5,3,4,8],
[1,9,8,3,4,2,5,6,7],

[8,5,9,7,6,1,4,2,3],
[4,2,6,8,5,3,7,9,1],
[7,1,3,9,2,4,8,5,6],

[9,6,1,5,3,7,2,8,4],
[2,8,7,4,1,9,6,3,5],
[3,4,5,2,8,6,1,7,9]
];

const board = document.getElementById("board");

function createBoard() {
    board.innerHTML = "";
    for (let i = 0; i < 9; i++) {
        let row = document.createElement("tr");
        for (let j = 0; j < 9; j++) {
            let cell = document.createElement("td");
            let input = document.createElement("input");

            if (puzzle[i][j] !== 0) {
                input.value = puzzle[i][j];
                input.disabled = true;
                input.classList.add("fixed");
            } else {
                input.oninput = () => validate(input, i, j);
            }

            input.maxLength = 1;

            if ((i+1)%3===0 && (j+1)%3===0)
                cell.classList.add("box");

            cell.appendChild(input);
            row.appendChild(cell);
        }
        board.appendChild(row);
    }
}

function validate(input, r, c) {
    if (!/[1-9]/.test(input.value)) {
        input.value = "";
        return;
    }
    if (parseInt(input.value) !== solution[r][c]) {
        input.classList.add("wrong");
        input.classList.remove("correct");
    } else {
        input.classList.add("correct");
        input.classList.remove("wrong");
    }
}

function checkWin() {
    let inputs = document.querySelectorAll("input");
    for (let i = 0; i < inputs.length; i++) {
        if (!inputs[i].disabled && !inputs[i].classList.contains("correct")) {
            alert("‚ùå Sudoku not solved correctly!");
            return;
        }
    }
    alert("üéâ Congratulations! You solved the Sudoku!");
}

function resetGame() {
    createBoard();
}

createBoard();
